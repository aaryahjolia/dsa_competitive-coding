int oppar(int app)

}
int neecha(int app)
{
    int ji;
    ji=app-1;

    ji=ji%10;

}

{

    while(kk--)
    {
        cin>>mm;
        for(ioi=0;ioi<mm;ioi++)

            tt=v[joj];
            for(ioi=0;ioi<bada;ioi++)
