## Code solution :

```cpp

#include <bits/stdc++.h>
using namespace std;
int main(){
    // Taking inputs
    int a,b,c,s1=0,s2=0;
    cin>>a>>b>>c;
    
    // Calculating expression with max value in the order a,b,c 
    if(a+b>a*b) s1+=(a+b);
    else s1+=a*b;
    
    if (s1+c>s1*c) s1+=c;
    else s1*=c;
    
    // Calculating expression with max value in the order c,b,a
    if (c+b>c*b) s2+=(c+b);
    else s2+=c*b;
    
    if (s2+a>s2*a) s2+=a;
    else s2*=a;
    
    // Printing max value among both sequence
    cout<<max(s1,s2)<<endl;
    return 0;
}

```
