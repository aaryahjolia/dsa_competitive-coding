## Code solution :

```cpp

#include <bits/stdc++.h>
using namespace std;
int main(){
    // Taking inputs
    int n, m;
    cin >> n >> m;
    int a[m];
    for (int i = 0; i < m; i++){
        cin >> a[i];
    }
    // Sorting in ascending order
    sort(a, a + m);
    // Initialising ans with max value to find minimum value
    int ans = INT_MAX;
    // To check every n integer's difference and storing the minimum value among them all and printing
    for (int i = 0; i + n <= m; i++){
        ans = min(ans, a[i + n - 1] - a[i]);
    }
    cout << ans;
    return 0;
}

```
