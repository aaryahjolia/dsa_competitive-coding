## Code solution :

```cpp

#include <bits/stdc++.h>
using namespace std;
int main(){
    // Taking inputs
    int t;
    cin>>t;
    while(t--){
        // Taking inputs
        long long int a,b,k;
        cin>>a>>b;
        //Swapping a and b
        if (a<b){
            k=b;
            b=a;
            a=k;
        }   
        k=a-b;
        if (abs(k)==1) cout<<1<<" "<<0<<endl;
        else if(k==0) cout<<0<<" "<<0<<endl;
        else cout<<k<<" "<<min(a%k,k-a%k)<<endl;
    }
    return 0;
}

```
