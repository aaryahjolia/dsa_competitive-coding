## Code solution :

```cpp

#include <bits/stdc++.h>
using namespace std;
int main(){
    //Taking n and array input
    int n;
    cin >> n;
    int a[n];
    for (int i = 0; i < n; i++){
        cin >> a[i];
    }
    // Sorting in descending number
    sort(a, a + n, greater<int>());
    // Storing prefix sum in same array
    for (int i = 1; i < n; i++){
        a[i] = a[i] + a[i - 1];
    }
    // Checking at which point the sum of money is greater than rest of it
    for (int i = 0; i < n; i++){
        if (a[i] > (a[n - 1] - a[i])){
            cout << (i + 1) << endl;
            break;
        }
    }
    return 0;
}
```
