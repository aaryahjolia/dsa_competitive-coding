## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // Input
    string a,b;
    cin>>a>>b;
    // reversing string a
    for(int i=0;i<(a.length()/2);i++){
        char temp=a[i];
        a[i]=a[a.length()-1-i];
        a[a.length()-1-i]=temp;
    }
    // if equal, yes
    if(a==b)cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
return 0;
}

```
