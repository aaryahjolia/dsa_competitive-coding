#include <iostream>
using namespace std;
 
int main() 
{
	int t; 
	cin>>t;
	while(t--) //test cases
	{
		int n; 
		cin>>n;
		int arr[n];
		
		for(int i=0;i<n;i++)
		{
		    cin>>arr[i];
		}
		
		for(int i=0;i<n;i++)
		{
			int x; 
			cin>>x;
			string str;
			cin>>str;
			
			for(int j=0;j<x;j++)
			{
				if(str[j]=='U') //checking the string character with 'U'
				{
				if(arr[i]==0)
				arr[i]=9;
				else
				arr[i]--;
				}
				else if(str[j]=='D')  //checking the string character with 'D'
				{
					if(arr[i]==9)
					arr[i]=0;
					else
					arr[i]++;
				}
			}
		}
		for(int i=0;i<n;i++)
		{
		    cout<<arr[i]<<" "; //printing the new array
		}
	
		cout<<"\n";
	}
	return 0;
}
