https://codeforces.com/problemset/problem/266/B

## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // Inputs
    int n,t;
    cin>>n>>t;
    string s;
    cin>>s;
    // nested loops
    for(int i=0;i<t;i++){
        for(int j=0;j<s.length();j++){
            // if B is ahead of G, interchange position
            if(s[j]=='B' && s[j+1]=='G'){
                s[j]='G';
                s[j+1]='B';
                j++;
            }
        }
    }
    cout<<s<<endl;
return 0;
}

```
