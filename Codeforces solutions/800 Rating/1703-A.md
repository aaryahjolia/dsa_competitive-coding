## Code solution :

```cpp

#include <iostream>
 
using namespace std;
 
int main()
{
    string s;
    int t,i,c;
    cin>>t;
    while(t--)
    {
        c=0;
        cin>>s;
        for(i=0;i<3;i++)
    {
        if(s[i]<92)
        s[i]+=32;
    }
    if(s[0]=='y')
    c++;
    if(s[1]=='e')
    c++;
    if(s[2]=='s')
    c++;
    
    if(c==3)
    cout<<"YES\n";
    else
    cout<<"NO\n";
    }
    return 0;
}
```
