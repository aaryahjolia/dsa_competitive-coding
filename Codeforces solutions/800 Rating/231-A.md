## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,main_ans=0;
    cin>>n;
    while(n--){
        int a,b,c;
        cin>>a>>b>>c;

        //1st way - if combined score is >=2
        // int s=a+b+c;
        // if(s>=2)main_ans++;

        //2nd way - If any of 2 has score=2
        if(a+b==2 || b+c==2 || a+c==2)main_ans++;
    }
    cout<<main_ans<<endl;
return 0;
}
```
