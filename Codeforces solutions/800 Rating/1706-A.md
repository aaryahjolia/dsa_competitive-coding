#include <bits/stdc++.h>
using namespace std;

int main()
{
    // Taking inputs
    int t;
    cin>>t;
    while (t--){
        // Taking inputs
        int n,m,mi,ma;
        cin>>n>>m;
        int a[n];
        for(int i=0;i<n;i++) cin>>a[i];
        // Creating array with m elements of 'B'
        char s[m];
        fill_n(s,m,'B');
        // Replacing B with A for ai-th or the (m+1âˆ’ai)-th element
        for(int i=0;i<n;i++){
            mi=min(a[i]-1,m-a[i]);
            ma=max(a[i]-1,m-a[i]);
            if (s[mi]=='B') s[mi]='A';
            else s[ma]='A';
        }
        // Printing Lexicographically smallest string
        for(int i=0;i<m;i++) cout<<s[i];
        cout<<endl;
    }
    return 0;
}
