## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    cin>>s;
    int sum=0;
    // Placing '.' in place of duplicate chars
    for(int i=0;i<s.length();i++){
        for(int j=i+1;j<s.length();j++){
            if(s[i]==s[j])s[j]='.';
        }
    }
    // counting char count (not of '.')
    for(int i=0;i<s.length();i++){
        if(s[i]!='.')sum++;
    }
    // Relevant message
    if(sum%2==0)cout<<"CHAT WITH HER!"<<endl;
    else cout<<"IGNORE HIM!"<<endl;
return 0;
}
```
