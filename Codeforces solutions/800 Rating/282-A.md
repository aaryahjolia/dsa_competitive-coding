## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // Input
    int n,x=0;
    cin>>n;
    while(n--){
        string a;
        cin>>a;
        // If first char = '+', then ++x, else --x
        if(a[0]=='+')++x;
        else if(a[0]=='-')--x;
        // If first char=X or x, but second char = '+',then x++, else x--
        else if(a[0]=='X' || a[0]=='x'){
            if(a[1]=='+')x++;
            else if(a[1]=='-')x--;
        }
    }
    cout<<x<<endl;
return 0;
}

```
