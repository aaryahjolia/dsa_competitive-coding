## Code solution :

```cpp

#include <bits/stdc++.h>
using namespace std;

int main()
{
    // Taking inputs
    int t;
    scanf("%d",&t);
    while (t--){
        // Taking inputs
        int a;
        cin>>a;
        // Printing division number according to given ratings
        if (a<=1399) cout<<"Division 4"<<endl;
        else if (a>=1400 && a<=1599) cout<<"Division 3"<<endl;
        else if (a>=1600 && a<=1899) cout<<"Division 2"<<endl;
        else cout<<"Division 1"<<endl;
    }
    return 0;
}
```
