## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // Input
    int w;
    cin>>w;
    // Checking if any 2 integer upto w makes addition equal to w and printing msg.
    for(int i=2;i<w;i=i+2){
        for(int j=2;j<w;j=j+2){
            if(i+j==w){
                cout<<"YES"<<endl;
                goto label;
            }
        }
    }
    
    cout<<"NO"<<endl;
    label:
return 0;
}
```
