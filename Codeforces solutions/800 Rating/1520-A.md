## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // Input
    int t;
    cin>>t;
    while(t--){
        int a;
        cin>>a;
        string s;
        cin>>s;
        bool z=false;
        // nested loop and checking if immediate number is diff but other num after immediate is same and printing
        for(int i=0;i<a;i++){
            for(int j=i+1;j<a;j++){
                if(s[i]==s[j] && s[i]!=s[i+1]){
                    z=true;
                    goto label;
                }
            }
        }
        label:;
        if(z==false)cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
return 0;
}
```
