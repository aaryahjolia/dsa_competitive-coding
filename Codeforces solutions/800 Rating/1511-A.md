## Code solution :

```cpp

#include<iostream>
using namespace std;
int main(){
    // Input
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int a[n],sum=0;
        // Array input
        for(int i=0;i<n;i++){
            cin>>a[i];
        }
        // If value is 1 or 3, increment sum and print
        for(int i=0;i<n;i++){
            if(a[i]==1)sum++;
            if(a[i]==3)sum++;
        }
        cout<<sum<<endl;
    }
return 0;
}

```
