## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // Input
    int n;
    cin>>n;
    int a[n],b[n];
    // Taking input array and setting index of b as number of a and printing
    for(int i=0;i<n;i++){
        cin>>a[i];
        b[a[i]-1]=i+1;
    }
    for(int i=0;i<n;i++){
        cout<<b[i]<<" ";
    }
return 0;
}

```
