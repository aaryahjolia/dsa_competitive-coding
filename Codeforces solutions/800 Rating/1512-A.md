## Code solution :

```cpp

#include<iostream>
using namespace std;
int main(){
    // Input
    int t;
    cin>>t;
    while(t--){
        int n,k=0;
        cin>>n;
        int arr[100];
        // Array input
        for(int i=0;i<n;i++){
            cin>>arr[i];
        }
        // Calculating same numbers, storing in k
        if(arr[0]==arr[1])k=arr[0];
        if(arr[0]==arr[2])k=arr[0];
        // Checking if number not equals k and printing
        for(int i=0;i<n;i++){
            if(arr[i]!=k){
                cout<<i+1<<endl;
                break;
            }
        }
    }
return 0;
}

```
