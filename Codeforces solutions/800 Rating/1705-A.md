## Code solution :

```cpp

#include <bits/stdc++.h>
using namespace std;

int main()
{
    // Taking inputs
    int t;
    scanf("%d",&t);
    while (t--){
        // Taking inputs
        int n,x;
        cin>>n>>x;
        int a[2*n];
        for(int i=0;i<2*n;i++) cin>>a[i];

        sort(a,a+2*n);
        bool flag=true;

        // Comparing first half with second half heights to ensure min difference
        for(int i=0;i<n;i++){
            if ((a[i+n]-a[i])<x){
                cout<<"NO"<<endl;
                flag=false;
                break;
            }
        } if (flag) cout<<"YES"<<endl;
    }
    return 0;
}

```
