## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // Input
    int n;
    cin>>n;
    string s;
    cin>>s;
    int a=0,d=0;
    // counting anton and darik wins
    for(int i=0;i<s.length();i++){
        if(s[i]=='A')a++;
        else d++;
    }
    // appropriate msg according to count
    if(a>d)cout<<"Anton"<<endl;
    else if(d>a)cout<<"Danik"<<endl;
    else cout<<"Friendship"<<endl;
return 0;
}

```
