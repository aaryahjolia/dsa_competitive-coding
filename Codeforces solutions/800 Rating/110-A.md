## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    long long int a,sum=0;
    cin>>a;
    // while number becomes zero
    while(a!=0){
        // counting number of 4 or 7 and removing last digit
        if(a%10==4 || a%10==7)sum++;
        a=a/10;
    }
    // if count=4 or 7, then yes
    if(sum==4 || sum==7)cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
return 0;
}

```
