## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // input
    int x,result=0;
    cin>>x;
    // loop till distance become zero
    while(x!=0){
        // if more than 5 left, run 5 steps
        if(x>5){
            result++;
            x=x-5;
        }
        // else we have reached in one count
        else{
            result++;
            x=0;
        }
    }
    cout<<result<<endl;
return 0;
}

```
