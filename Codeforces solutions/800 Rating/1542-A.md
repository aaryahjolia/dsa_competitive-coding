## Code solution :

```cpp

#include <bits/stdc++.h>
using namespace std;
int main()
{
    // Taking inputs
    int t;
    cin>>t;
    while (t--){
        int e=0,o=0,n,c;
        cin>>n;;
        n=2*n;
        // Calculate number of even and odd elements  
        while (n--){
            cin>>c;
            if(c%2==0)
                e+=1;
            else
                o+=1;
        }
        // If number of even elements equal to odd odd elements,it makes a pair
        if(e==o) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
    }
    return 0;
}

```
