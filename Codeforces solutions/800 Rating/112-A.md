## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // Input
    string a,b;
    cin>>a>>b;
    // converting both in lowercase
    transform(a.begin(),a.end(),a.begin(),::tolower);
    transform(b.begin(),b.end(),b.begin(),::tolower);
    // compare result output
    cout<<a.compare(b)<<endl;
return 0;
}

```
