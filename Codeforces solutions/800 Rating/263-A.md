https://codeforces.com/problemset/problem/263/A

## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // Input
    int a[5][5],x,y,sum=0;
    for(int i=0;i<5;i++){
        for(int j=0;j<5;j++){
            cin>>a[i][j];
            // Storing index of '1'
            if(a[i][j]==1){
                x=i;
                y=j;
            }
        }
    }
    // Calculating how much that index is away from index 2 (Mid index)
    sum=abs(x-2)+abs(y-2);
    cout<<sum<<endl;
return 0;
}

```
